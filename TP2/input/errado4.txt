{
	float [4] numeros;
	float temp;
	int tamanho;
	int j;
	int i;

	numeros[0] = 10;
	numeros[1] = -3.9;
	numeros[2] = 2592;
	numeros[3] = 0,1;

	tamanho = 4;

	i = 0;
	while ( i < tamanho ) {
		j = i;

		while (j > 0 && numeros[j] < numeros[j - 1]) {
			temp = numeros[j];
			numeros[j] = numeros[j - 1];
			numeros[j - 1] = temp;
			j = j -1;
		}

		i = i + 1;
	}
}
